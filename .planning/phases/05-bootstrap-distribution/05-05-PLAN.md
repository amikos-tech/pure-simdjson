```go
package main

import (
    "context"
    "fmt"
    "io"
    "strings"

    "github.com/amikos-tech/pure-simdjson/internal/bootstrap"
    "github.com/spf13/cobra"
)

func newFetchCmd() *cobra.Command {
    var (
        allPlatforms bool
        targets      []string
        dest         string
        version      string
        mirror       string
    )
    cmd := &cobra.Command{
        Use:   "fetch",
        Short: "Download artifacts to cache or --dest",
        RunE: func(cmd *cobra.Command, args []string) error {
            return runFetch(cmd.Context(), allPlatforms, targets, dest, version, mirror, cmd.ErrOrStderr())
        },
    }
    cmd.Flags().BoolVar(&allPlatforms, "all-platforms", false, "fetch for all 5 supported platforms")
    cmd.Flags().StringArrayVar(&targets, "target", nil, "fetch for specific os/arch, e.g. linux/amd64 (repeatable)")
    cmd.Flags().StringVar(&dest, "dest", "", "destination directory (default: OS user cache)")
    cmd.Flags().StringVar(&version, "version", "", "library version (default: embedded Version constant)")
    cmd.Flags().StringVar(&mirror, "mirror", "", "override R2 base URL (same as PURE_SIMDJSON_BINARY_MIRROR)")
    return cmd
}

func runFetch(ctx context.Context, allPlatforms bool, targets []string, dest, version, mirror string, errW io.Writer) error {
    var platforms [][2]string

    if allPlatforms {
        platforms = bootstrap.SupportedPlatforms
    } else if len(targets) > 0 {
        for _, t := range targets {
            parts := strings.SplitN(t, "/", 2)
            if len(parts) != 2 {
                return fmt.Errorf("invalid target %q: expected os/arch format", t)
            }
            platforms = append(platforms, [2]string{parts[0], parts[1]})
        }
    } else {
        // Default: fetch for current platform only.
        // Use runtime.GOOS/GOARCH — or rely on BootstrapSync default.
        platforms = nil // empty means BootstrapSync uses runtime.GOOS/GOARCH default
    }

    buildOpts := func(goos, goarch string) []bootstrap.BootstrapOption {
        var opts []bootstrap.BootstrapOption
        if dest != "" {
            opts = append(opts, bootstrap.WithDest(dest))
        }
        if version != "" {
            opts = append(opts, bootstrap.WithVersion(version))
        }
        if mirror != "" {
            opts = append(opts, bootstrap.WithMirror(mirror))
        }
        if goos != "" {
            opts = append(opts, bootstrap.WithTarget(goos, goarch))
        }
        return opts
    }

    if len(platforms) == 0 {
        // No explicit target: fetch current platform.
        fmt.Fprintln(errW, "fetching current platform artifact...")
        return bootstrap.BootstrapSync(ctx, buildOpts("", "")...)
    }

    for _, p := range platforms {
        goos, goarch := p[0], p[1]
        fmt.Fprintf(errW, "fetching %s/%s...\n", goos, goarch)
        if err := bootstrap.BootstrapSync(ctx, buildOpts(goos, goarch)...); err != nil {
            return fmt.Errorf("fetch %s/%s: %w", goos, goarch, err)
        }
    }
    return nil
}
```

```go
package main

import (
    "crypto/sha256"
    "encoding/hex"
    "errors"
    "fmt"
    "io"
    "os"
    "runtime"

    "github.com/amikos-tech/pure-simdjson/internal/bootstrap"
    "github.com/spf13/cobra"
)

func newVerifyCmd() *cobra.Command {
    return &cobra.Command{
        Use:   "verify",
        Short: "Re-verify SHA-256 of cached artifacts against embedded checksums",
        RunE: func(cmd *cobra.Command, args []string) error {
            return runVerify(cmd.ErrOrStderr())
        },
    }
}

func runVerify(errW io.Writer) error {
    goos, goarch := runtime.GOOS, runtime.GOARCH
    cachePath := bootstrap.CachePath(goos, goarch)
    key := bootstrap.ChecksumKey(bootstrap.Version, goos, goarch)
    expected, ok := bootstrap.Checksums[key]
    if !ok {
        return fmt.Errorf("%w: %s", bootstrap.ErrNoChecksum, key)
    }

    f, err := os.Open(cachePath)
    if err != nil {
        return fmt.Errorf("open %s: %w", cachePath, err)
    }
    defer f.Close()

    h := sha256.New()
    if _, err := io.Copy(h, f); err != nil {
        return fmt.Errorf("hash %s: %w", cachePath, err)
    }
    got := hex.EncodeToString(h.Sum(nil))

    if got != expected {
        fmt.Fprintf(errW, "FAIL %s: expected %s, got %s\n", cachePath, expected, got)
        return bootstrap.ErrChecksumMismatch
    }
    fmt.Printf("PASS %s\n", cachePath)
    return nil
}

// ensure errors package used for tests
var _ = errors.New
```

```go
import (
    "fmt"
    "os"

    "github.com/amikos-tech/pure-simdjson/internal/bootstrap"
    "github.com/spf13/cobra"

func newPlatformsCmd() *cobra.Command {
    return &cobra.Command{
        Use:   "platforms",
        Short: "List supported platforms and local cache presence",
        RunE: func(cmd *cobra.Command, args []string) error {
            return runPlatforms()
        },
    }
}

func runPlatforms() error {
    for _, p := range bootstrap.SupportedPlatforms {
        goos, goarch := p[0], p[1]
        cachePath := bootstrap.CachePath(goos, goarch)
        indicator := "✗ missing"
        if _, err := os.Stat(cachePath); err == nil {
            indicator = "✓ cached"
        }
        fmt.Printf("%-20s %s\n", goos+"/"+goarch, indicator)
    }
    return nil
}
```

```go
package main

import (
    "context"
    "crypto/sha256"
    "encoding/hex"
    "io"
    "net/http"
    "net/http/httptest"
    "os"
    "path/filepath"
    "strings"
    "testing"

    "github.com/amikos-tech/pure-simdjson/internal/bootstrap"
)

// TestFetchCmd verifies that runFetch with --all-platforms downloads all 5
// artifacts to --dest. Uses httptest.Server as the mirror.
func TestFetchCmd(t *testing.T) {
    fakeBody := []byte("fake-library-bytes-for-fetch-test")
    fakeHex := func(b []byte) string {
        h := sha256.New()
        h.Write(b)
        return hex.EncodeToString(h.Sum(nil))
    }(fakeBody)
    _ = fakeHex // used when Checksums lookup happens

    // Build a mux serving all 5 platform artifacts.
    mux := http.NewServeMux()
    var hits int
    for _, p := range bootstrap.SupportedPlatforms {
        goos, goarch := p[0], p[1]
        // path: /v<Version>/<goos>-<goarch>/<libname>
        libName := filepath.Base(bootstrap.ChecksumKey(bootstrap.Version, goos, goarch))
        urlPath := "/v" + bootstrap.Version + "/" + goos + "-" + goarch + "/" + libName
        urlPath = strings.ReplaceAll(urlPath, `\`, "/") // windows safety
        mux.HandleFunc(urlPath, func(w http.ResponseWriter, r *http.Request) {
            hits++
            _, _ = w.Write(fakeBody)
        })
    }
    mux.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        http.NotFound(w, r)
    })
    srv := httptest.NewServer(mux)
    t.Cleanup(srv.Close)

    destDir := t.TempDir()
    t.Setenv("PURE_SIMDJSON_DISABLE_GH_FALLBACK", "1")
    t.Setenv("PURE_SIMDJSON_BINARY_MIRROR", srv.URL)

    // Temporarily replace Checksums with fake values so verification passes.
    origChecksums := bootstrap.Checksums
    fakeChecksums := make(map[string]string)
    for _, p := range bootstrap.SupportedPlatforms {
        k := bootstrap.ChecksumKey(bootstrap.Version, p[0], p[1])
        fakeChecksums[k] = fakeHex
    }
    bootstrap.Checksums = fakeChecksums
    t.Cleanup(func() { bootstrap.Checksums = origChecksums })

    err := runFetch(context.Background(), true, nil, destDir, "", srv.URL, io.Discard)
    if err != nil {
        t.Fatalf("runFetch --all-platforms: %v", err)
    }
    if hits != 5 {
        t.Fatalf("expected 5 downloads (one per platform), got %d", hits)
    }
    // Verify all 5 artifacts exist at destDir.
    for _, p := range bootstrap.SupportedPlatforms {
        // CachePath uses actual OS cache dir; --dest overrides to destDir.
        // Check that some file exists under destDir for each platform.
        found := false
        err := filepath.Walk(destDir, func(path string, info os.FileInfo, err error) error {
            if err != nil { return err }
            if !info.IsDir() { found = true }
            return nil
        })
        if err != nil || !found {
            t.Errorf("no artifact found for %s/%s under %s", p[0], p[1], destDir)
        }
    }
}
```
