```go
package bootstrap

import (
	"errors"
	"fmt"
	"os"
	"path/filepath"
	"time"
)

const (
	lockAcquireTimeout = 2 * time.Minute
	lockRetryInterval  = 200 * time.Millisecond
	lockLogInterval    = 5 * time.Second

	// cacheDirEnvVar allows callers (CI runners with ephemeral HOME) to override
	// the OS user cache dir. L2 from 05-REVIEWS.md.
	cacheDirEnvVar = "PURE_SIMDJSON_CACHE_DIR"
)

// defaultCacheDir returns the base cache directory for pure-simdjson artifacts.
// Precedence (L2 + L6):
//  1. PURE_SIMDJSON_CACHE_DIR env var, if non-empty
//  2. os.UserCacheDir() + "/pure-simdjson"
//  3. os.TempDir() + "/pure-simdjson-<uid>" with 0700 perms (L6 ŌĆö private subdir,
//     never the bare TempDir path)
// Source: pure-onnx@v0.0.1/ort/bootstrap.go lines 1370-1383, adapted for L2/L6.
func defaultCacheDir() string {
	if env := os.Getenv(cacheDirEnvVar); env != "" {
		return env
	}
	base, err := os.UserCacheDir()
	if err == nil && base != "" {
		return filepath.Join(base, "pure-simdjson")
	}
	// L6: fallback on systems where UserCacheDir fails (rare; headless CI without
	// HOME). Use a UID-scoped subdirectory under TempDir with 0700 perms so the
	// cache remains non-world-writable (DIST-05 spirit preserved).
	uid := os.Getuid() // POSIX; on Windows Getuid returns -1
	sub := fmt.Sprintf("pure-simdjson-%d", uid)
	path := filepath.Join(os.TempDir(), sub)
	// Best-effort create with 0700; errors are surfaced later at MkdirAll call sites.
	_ = os.MkdirAll(path, 0700)
	return path
}

// CachePath returns the absolute path where the artifact for goos/goarch is stored.
// Layout: <cacheDir>/v<Version>/<goos>-<goarch>/<libname> (D-07).
// Called from library_loading.go::resolveLibraryPath to check cache presence (D-04).
func CachePath(goos, goarch string) string {
	return artifactCachePath(defaultCacheDir(), Version, goos, goarch)
}

func artifactCachePath(cacheDir, version, goos, goarch string) string {
	return filepath.Join(cacheDir,
		"v"+version,
		goos+"-"+goarch,
		platformLibraryName(goos))
}

// withProcessFileLock acquires an exclusive flock on lockPath, calls fn, releases lock.
// Polling: 200ms interval, 2-minute timeout.
// Source: pure-onnx@v0.0.1/ort/bootstrap.go lines 1283-1334 (verbatim, rename package).
func withProcessFileLock(lockPath string, fn func() error) (err error) {
	if fn == nil {
		return fmt.Errorf("lock callback is nil")
	}
	if err := os.MkdirAll(filepath.Dir(lockPath), 0700); err != nil {
		return fmt.Errorf("create lock dir: %w", err)
	}
	file, err := os.OpenFile(lockPath, os.O_CREATE|os.O_RDWR, 0600)
	if err != nil {
		return fmt.Errorf("open lock file %q: %w", lockPath, err)
	}

	start := time.Now()
	nextLogAt := start.Add(lockLogInterval)
	for {
		lockErr := lockFile(file)
		if lockErr == nil {
			break
		}
		if !isLockWouldBlock(lockErr) {
			acquireErr := fmt.Errorf("acquire lock %q: %w", lockPath, lockErr)
			_ = file.Close()
			return acquireErr
		}
		waited := time.Since(start)
		if waited >= lockAcquireTimeout {
			_ = file.Close()
			return fmt.Errorf("timed out acquiring lock %q after %s", lockPath, lockAcquireTimeout)
		}
		if time.Now().After(nextLogAt) {
			nextLogAt = time.Now().Add(lockLogInterval)
		}
		time.Sleep(lockRetryInterval)
	}

	defer func() {
		unlockErr := unlockFile(file)
		closeErr := file.Close()
		err = errors.Join(err, unlockErr, closeErr)
	}()

	return fn()
}

// atomicInstall renames tmpPath to finalPath (same filesystem ŌĆö os.CreateTemp in
// cacheDir guarantees same-FS). Removes tmpPath on failure.
// Critical: tmpPath MUST be in the same directory as finalPath for atomic rename.
func atomicInstall(tmpPath, finalPath string) error {
	if err := os.MkdirAll(filepath.Dir(finalPath), 0700); err != nil {
		return fmt.Errorf("create artifact dir: %w", err)
	}
	if err := os.Rename(tmpPath, finalPath); err != nil {
		_ = os.Remove(tmpPath)
		return fmt.Errorf("atomic rename to %s: %w", finalPath, err)
	}
	return nil
}
```

**M2 ŌĆö Failure memoization (locked design):**

```go
const bootstrapFailureTTL = 30 * time.Second

type bootstrapFailureCache struct {
	mu        sync.Mutex
	lastErr   error
	expiresAt time.Time
}

var globalBootstrapFailureCache bootstrapFailureCache

// peek returns a non-nil cached error if one is currently valid; nil otherwise.
func (c *bootstrapFailureCache) peek() error {
	c.mu.Lock()
	defer c.mu.Unlock()
	if c.lastErr != nil && time.Now().Before(c.expiresAt) {
		return c.lastErr
	}
	return nil
}

// record stores err as the most recent failure, expiring after bootstrapFailureTTL.
func (c *bootstrapFailureCache) record(err error) {
	c.mu.Lock()
	defer c.mu.Unlock()
	c.lastErr = err
	c.expiresAt = time.Now().Add(bootstrapFailureTTL)
}

// clear resets the cache on successful bootstrap.
func (c *bootstrapFailureCache) clear() {
	c.mu.Lock()
	defer c.mu.Unlock()
	c.lastErr = nil
	c.expiresAt = time.Time{}
}
```

`BootstrapSync` checks `peek()` at entry. If it returns a non-nil error, wrap with a
short explanatory prefix (`fmt.Errorf("bootstrap failed (memoized): %w", cached)`) and
return immediately. On network success, call `clear()`. On network failure, call
`record(err)` before returning. The caller's `ctx` is checked first ŌĆö if the context
is already cancelled, return `ctx.Err()` without consulting the cache.

**Important:** ErrChecksumMismatch (permanent) and ErrAllSourcesFailed both get
memoized. The 30-second TTL is short enough that a user retrying after env-var fix
will re-attempt. TTL is NOT configurable in v0.1 ŌĆö keep it simple.

5. `isRetryable(statusCode int, headers http.Header, bodySnippet string) bool` ŌĆö D-16: 408,429,5xx retryable; 403 with rate-limit body-sniff retryable (Pattern 6)
6. `isBootstrapRedirectPolicyError(err error) bool` ŌĆö checks if err is a redirect policy error
7. `downloadWithRetry(ctx context.Context, cfg bootstrapConfig, primaryURL, fallbackURL string) (tmpPath, hexDigest string, err error)` ŌĆö D-15 sequential fallback loop: 3 R2 attempts, then 3 GH attempts (unless disableGH or fallbackURL empty); Full-Jitter sleep; break on permanent error
8. `downloadOnce(ctx context.Context, cfg bootstrapConfig, rawURL, cacheDir string) (tmpPath, hexDigest string, err error)` ŌĆö http.NewRequestWithContext, User-Agent header, status check, os.CreateTemp(cacheDir), io.Copy(io.MultiWriter(f, sha256Hasher), io.LimitReader(body, 128MiB+1)), hex encode
9. `downloadAndVerify(ctx context.Context, cfg bootstrapConfig, cachePath, goos, goarch string) error` ŌĆö calls downloadWithRetry, verifies hexDigest against Checksums[ChecksumKey(...)], ErrNoChecksum if missing entry, ErrChecksumMismatch if mismatch, then atomicInstall(tmpPath, cachePath)

Key invariants:
- `os.CreateTemp(cacheDir, "pure-simdjson-*.tmp")` ŌĆö cacheDir NOT os.TempDir() (pitfall #3)
- `downloadAndVerify` returns `markPermanentBootstrapError(ErrChecksumMismatch)` on mismatch ŌĆö no retry
- `downloadWithRetry` builds URL slice: [r2URL] then optionally [ghURL] based on disableGH flag
- `ghURL` is computed via `githubArtifactURL(cfg.githubBaseURL, version, goos, goarch)` ŌĆö when `cfg.githubBaseURL` is empty, the helper uses `defaultGitHubBaseURL`. This enables test injection (M3).
- Retry loop breaks immediately on `isPermanentBootstrapError(err)` ŌĆö 404 from R2 is NOT permanent (falls through to GH), but 404 from the final GH attempt bubbles up as ErrAllSourcesFailed
- Per D-17: "404 against R2 does fall through to GH; checksum mismatch does not retry anywhere" ŌĆö implement by NOT marking R2 404 as permanent in the R2 loop, and marking checksum mismatch explicitly as permanent in `downloadAndVerify`
