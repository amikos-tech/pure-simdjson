    Core function (per RESEARCH Pattern 5):
    ```python
    def parse_benchstat_for_regressions(text, threshold_pct, p_max):
        flagged = []
        seen_any_sec_op_row = False
        current_metric = None
        for line in text.splitlines():
            header = METRIC_HEADER_RE.search(line)
            if header:
                current_metric = header.group(1)
                continue
            if not ROW_PREFIX_RE.match(line):
                continue                       # ignores geomean and headers (Pitfall 2)
            if current_metric != "sec/op":
                continue                       # non-time metrics: +B/s is faster, not slower
            seen_any_sec_op_row = True
            if "~" in line:
                continue                       # benchstat sentinel (Pitfall 5)
            m = DELTA_RE.search(line)
            if not m:
                continue
            sign, pct_str, p_str = m.groups()
            if sign != "+":
                continue                       # faster, not slower
            pct = float(pct_str); p = float(p_str)
            if pct >= threshold_pct and p < p_max:
                row_name = line.strip().split(maxsplit=1)[0]
                flagged.append({"row": row_name, "delta_pct": pct,
                                "p_value": p, "raw_line": line.strip()})
        if not seen_any_sec_op_row:
            raise EvidenceError("no sec/op benchmark rows found in benchstat output")
        return flagged
    ```
